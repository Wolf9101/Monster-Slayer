// ── MAZE ─────────────────────────────────────────────────────────────────────
var MG=[],MS=9;
function mzSz(){return Math.min(21,9+(G.mzLv-1)*4);}
function mzSzOdd(n){return n%2===0?n+1:n;}

function buildMaze(){
  MS=mzSzOdd(mzSz());
  // Init all walls
  MG=[];
  var r,c;
  for(r=0;r<MS;r++){MG[r]=[];for(c=0;c<MS;c++)MG[r][c]=1;}

  // Iterative maze carver
  MG[1][1]=0;
  var stack=[[1,1]];
  var dirs=[[0,2],[2,0],[0,-2],[-2,0]];
  while(stack.length>0){
    var cur=stack[stack.length-1];
    var cr=cur[0],cc=cur[1];
    // Shuffle dirs
    var sd=dirs.slice().sort(function(){return Math.random()-0.5;});
    var moved=false;
    for(var d=0;d<sd.length;d++){
      var nr=cr+sd[d][0],nc=cc+sd[d][1];
      if(nr>0&&nr<MS-1&&nc>0&&nc<MS-1&&MG[nr][nc]===1){
        MG[cr+sd[d][0]/2][cc+sd[d][1]/2]=0;
        MG[nr][nc]=0;
        stack.push([nr,nc]);
        moved=true;break;
      }
    }
    if(!moved)stack.pop();
  }

  // Set exit
  MG[MS-2][MS-2]=2;

  // Add traps (level 2+) — never on start or exit
  if(G.mzLv>=2){
    var traps=G.mzLv*2;
    for(var t=0;t<traps;t++){
      var tr=2+Math.floor(Math.random()*(MS-4));
      var tc=2+Math.floor(Math.random()*(MS-4));
      if(MG[tr][tc]===0&&!(tr===MS-2&&tc===MS-2))MG[tr][tc]=3;
    }
  }

  // Place Miniture near center
  var mid=Math.floor(MS/2);
  // Find open cell near center
  var placed=false;
  for(var dist=0;dist<MS&&!placed;dist++){
    for(var dr=-dist;dr<=dist&&!placed;dr++){
      for(var dc=-dist;dc<=dist&&!placed;dc++){
        var mr=mid+dr,mc=mid+dc;
        if(mr>0&&mr<MS-1&&mc>0&&mc<MS-1&&MG[mr][mc]===0&&!(mr===1&&mc===1)&&!(mr===MS-2&&mc===MS-2)){
          G.mzMR=mr;G.mzMC=mc;placed=true;
        }
      }
    }
  }

  G.mzMA=true;G.mzMHp=5000+G.mzLv*1000;
  G.mzBA=true;G.mzBHp=8000+G.mzLv*2000;
  G.mzX=1;G.mzY=1;G.mzWon=false;
}

function onTilt(e){
  if(!G.on||G.world!=='maze')return;
  var now=Date.now();if(now-lastTilt<320)return;
  var gx=e.gamma||0,gy=e.beta||0;
  if(Math.abs(gx)<8&&Math.abs(gy)<8)return;
  if(Math.abs(gx)>=Math.abs(gy))mzMv(gx<0?-1:1,0);
  else mzMv(0,gy<0?-1:1);
  lastTilt=now;
}

    DeviceOrientationEvent.requestPermission().then(function(s){

    }).catch(function(){msg('Tilt not available.',2000);});

    window.addEventListener('deviceorientation',onTilt);msg('📱 Tilt controls ON!',2000);
  }
}

  else if(G.world==='dungeon')drawDungeon();
