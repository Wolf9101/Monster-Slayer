// ── MAZE ──────────────────────────────────────────────────────────────────────
let mazeGrid=[],MS=9;
function mazeSize(){return Math.min(29,9+(G.mazeLevel-1)*4);}
function genMaze(){
  MS=mazeSize();if(MS%2===0)MS++;
  mazeGrid=[];
  for(let r=0;r<MS;r++){mazeGrid[r]=[];for(let c=0;c<MS;c++)mazeGrid[r][c]=1;}
  function carve(r,c){[[0,2],[2,0],[0,-2],[-2,0]].sort(()=>Math.random()-.5).forEach(([dr,dc])=>{let nr=r+dr,nc=c+dc;if(nr>0&&nr<MS-1&&nc>0&&nc<MS-1&&mazeGrid[nr][nc]===1){mazeGrid[r+dr/2][c+dc/2]=0;mazeGrid[nr][nc]=0;carve(nr,nc);}});}
  mazeGrid[1][1]=0;carve(1,1);
  mazeGrid[MS-2][MS-2]=2;
  for(let i=0;i<G.mazeLevel*4;i++){let r=1+Math.floor(Math.random()*(MS-2)),c=1+Math.floor(Math.random()*(MS-2));if(mazeGrid[r][c]===0&&!(r===1&&c===1)&&!(r===MS-2&&c===MS-2))mazeGrid[r][c]=1;}
  if(G.mazeLevel>=2)for(let i=0;i<G.mazeLevel*3;i++){let r=2+Math.floor(Math.random()*(MS-4)),c=2+Math.floor(Math.random()*(MS-4));if(mazeGrid[r][c]===0)mazeGrid[r][c]=3;}
  if(G.mazeLevel>=4)for(let i=0;i<G.mazeLevel*2;i++){let r=2+Math.floor(Math.random()*(MS-4)),c=2+Math.floor(Math.random()*(MS-4));if(mazeGrid[r][c]===0&&!(r===1&&c===1)&&!(r===MS-2&&c===MS-2))mazeGrid[r][c]=4;}
  G.mazeX=1;G.mazeY=1;
}
genMaze();

// ── MAIN LOOP ─────────────────────────────────────────────────────────────────
function loop(){
  G.tick++;
  let phase=(G.tick%G.dayLen)/G.dayLen;G.day=phase<0.58;
  if(G.wasDay&&!G.day)sndNight();G.wasDay=G.day;
  if(G.secretPetActive&&G.gameOn){secretPetTimer++;if(secretPetTimer>=180){secretPetTimer=0;secretPetTick();}}
  updatePlayerList();showEventBanner();
  cx.clearRect(0,0,W,H);
  if(G.flying)drawSky();
  else{
    let evIds=EVENTS.map(e=>e.id);
    if(G.world==='hub')drawHub();
    else if(G.world==='dungeon')drawDungeon();
    else if(G.world==='special')drawSpecial();
    else if(G.world==='maze')drawMaze();
    else if(evIds.includes(G.world))drawEventWorld(G.world);
    else drawHub();
  }
  if(G.world!=='maze'){drawMonsters();drawPlayer();}
  hud();
  requestAnimationFrame(loop);
}
