var AC=null;
function gac(){
  try{
    if(!AC)AC=new(window.AudioContext||window.webkitAudioContext)();
    if(AC.state==='suspended')AC.resume();
  }catch(e){}
  return AC;
}
function beep(f,t,d,v,dl){
  try{
    var ac=gac();if(!ac)return;
    var o=ac.createOscillator(),g=ac.createGain();
    o.connect(g);g.connect(ac.destination);
    o.type=t||'sine';o.frequency.value=f||440;
    var s=ac.currentTime+(dl||0);
    g.gain.setValueAtTime(v||.15,s);
    g.gain.exponentialRampToValueAtTime(.001,s+(d||.2));
    o.start(s);o.stop(s+(d||.2)+.02);
  }catch(e){}
}

var MG=[],MS=9;
function mzSz(){return Math.min(21,9+(G.mzLv-1)*4);}
function mzOdd(n){return n%2===0?n+1:n;}
function buildMaze(){
  MS=mzOdd(mzSz());
  MG=[];
  var r,c;
  for(r=0;r<MS;r++){MG[r]=[];for(c=0;c<MS;c++)MG[r][c]=1;}
  MG[1][1]=0;
  var stack=[[1,1]],dirs=[[0,2],[2,0],[0,-2],[-2,0]];
  while(stack.length>0){
    var cur=stack[stack.length-1],cr=cur[0],cc=cur[1];
    var sd=dirs.slice().sort(function(){return Math.random()-.5;}),moved=false;
    for(var d=0;d<sd.length;d++){
      var nr=cr+sd[d][0],nc=cc+sd[d][1];
      if(nr>0&&nr<MS-1&&nc>0&&nc<MS-1&&MG[nr][nc]===1){
        MG[cr+Math.round(sd[d][0]/2)][cc+Math.round(sd[d][1]/2)]=0;
        MG[nr][nc]=0;stack.push([nr,nc]);moved=true;break;
      }
    }
    if(!moved)stack.pop();
  }
  // BFS guarantee path
  (function(){
    var vis=[],prev=[];
    for(var i=0;i<MS;i++){vis[i]=[];prev[i]=[];for(var j=0;j<MS;j++){vis[i][j]=false;prev[i][j]=null;}}
    var q=[[1,1]];vis[1][1]=true;
    while(q.length){
      var p=q.shift(),pr=p[0],pc=p[1];
      if(pr===MS-2&&pc===MS-2){var c2=[pr,pc];while(c2){MG[c2[0]][c2[1]]=0;c2=prev[c2[0]][c2[1]];}return;}
      var dd=[[0,1],[1,0],[0,-1],[-1,0]];
      for(var i=0;i<dd.length;i++){
        var nr=pr+dd[i][0],nc=pc+dd[i][1];
        if(nr>=0&&nr<MS&&nc>=0&&nc<MS&&!vis[nr][nc]){vis[nr][nc]=true;prev[nr][nc]=[pr,pc];q.push([nr,nc]);}
      }
    }
    // fallback: open a straight path
    for(var i=1;i<MS-1;i++){MG[1][i]=0;MG[i][MS-2]=0;}
  })();
  MG[MS-2][MS-2]=2;
  // Traps level 2+
  if(G.mzLv>=2){
    for(var t=0;t<G.mzLv*2;t++){
      var tr=2+Math.floor(Math.random()*(MS-4)),tc=2+Math.floor(Math.random()*(MS-4));
      if(MG[tr][tc]===0&&!(tr===MS-2&&tc===MS-2))MG[tr][tc]=3;
    }
  }
  // Miniture near center
  var mid=Math.floor(MS/2),placed=false;
  for(var dist=0;dist<MS&&!placed;dist++){
    for(var dr=-dist;dr<=dist&&!placed;dr++){
      for(var dc=-dist;dc<=dist&&!placed;dc++){
        var mr=mid+dr,mc=mid+dc;
        if(mr>0&&mr<MS-1&&mc>0&&mc<MS-1&&MG[mr][mc]===0&&!(mr===1&&mc===1)&&!(mr===MS-2&&mc===MS-2)){G.mzMR=mr;G.mzMC=mc;placed=true;}
      }
    }
  }
  G.mzMA=true;G.mzMHp=3000+G.mzLv*500;
  G.mzBA=true;G.mzBHp=5000+G.mzLv*1000;
  G.mzX=1;G.mzY=1;G.mzWon=false;
}

function mzMv(dx,dy){
  if(!G.on||G.world!=='maze'||G.mzWon)return;
  gac();
  var nx=G.mzX+dx,ny=G.mzY+dy;
  if(nx<0||ny<0||nx>=MS||ny>=MS){showMsg('Edge of maze!');return;}
  var cell=MG[ny][nx];
  if(cell===1){beep(120,'square',.05,.1);showMsg('🧱 Wall! Try another direction.');return;}
  sMzMv();G.mzX=nx;G.mzY=ny;
  if(cell===3){sTrap();G.hp=Math.max(1,G.hp-50);showMsg('🪤 Trap! -50 HP!');if(G.hp<=0){die();return;}hud();}
  if(G.mzMA&&G.mzX===G.mzMC&&G.mzY===G.mzMR){showMsg('👺 MINITURE IS HERE!\nTap ⚔️ to fight!\n(Hits back 500 dmg!)');return;}
  if(G.mzX===MS-2&&G.mzY===MS-2){
    if(G.mzBA){showMsg('🔴 BLOODY CYCLOPS guards the exit!\nTap ⚔️ to fight it!');return;}
    else{mzWin();return;}
  }
  // Chase from lv2
  if(G.mzLv>=2&&G.mzMA){
    var dr=G.mzY-G.mzMR,dc=G.mzX-G.mzMC;
    var sr=Math.abs(dr)>=Math.abs(dc)?Math.sign(dr):0,sc2=sr===0?Math.sign(dc):0;
    var nr2=G.mzMR+sr,nc2=G.mzMC+sc2;
    if(nr2>0&&nr2<MS-1&&nc2>0&&nc2<MS-1&&MG[nr2][nc2]!==1){G.mzMR=nr2;G.mzMC=nc2;}
    if(G.mzMR===G.mzY&&G.mzMC===G.mzX){G.hp=Math.max(1,G.hp-200);sHit();showMsg('👺 Miniture caught you! -200 HP!\nTap ⚔️ now!');if(G.hp<=0)die();hud();}
  }
}

// Tilt
var lastTilt=0;
function onTilt(e){
  if(!G.on||G.world!=='maze')return;
  var now=Date.now();if(now-lastTilt<320)return;
  var gx=e.gamma||0,gy=e.beta||0;
  if(Math.abs(gx)<8&&Math.abs(gy)<8)return;
  if(Math.abs(gx)>=Math.abs(gy))mzMv(gx<0?-1:1,0);
  else mzMv(0,gy<0?-1:1);
  lastTilt=now;
}
function enableTilt(){
  if(typeof DeviceOrientationEvent!=='undefined'&&typeof DeviceOrientationEvent.requestPermission==='function'){
    DeviceOrientationEvent.requestPermission().then(function(s){
      if(s==='granted'){window.addEventListener('deviceorientation',onTilt);showMsg('📱 Tilt ON! Tilt iPad in maze!',2500);}
    }).catch(function(){showMsg('Tilt not available.',2000);});
  } else {
    window.addEventListener('deviceorientation',onTilt);showMsg('📱 Tilt controls ON!',2000);
  }
  closeScreen();
}

function drawMons(){
  for(var i=0;i<MONS.length;i++){
    var m=MONS[i];if(!m.alive||m.world!==G.world||G.world==='maze'||m.id===99)continue;
    if(G.tick%8===0){var dx=G.px-m.x,dy=G.py-m.y,d=Math.hypot(dx,dy),sp=G.day?1.6:3.4;if(d>60&&d<W*.5){m.x+=dx/d*sp;m.y+=dy/d*sp;}}
    var fs=m.mini?Math.floor(H*.055):Math.floor(H*.042);
    cx.font=fs+'px serif';cx.fillText(m.em,m.x-fs*.5,m.y+fs*.4);

function loop(){
  G.tick++;
  var ph=(G.tick%G.dlen)/G.dlen;G.day=ph<.58;
  if(G.wday&&!G.day)sNight();G.wday=G.day;
  spTimer++;if(spTimer>=180){spTimer=0;spTick();}
  mpTick++;if(mpTick>=120){mpTick=0;mpWrite();mpRead();}
  showEvBanner();
  cx.clearRect(0,0,W,H);
  if(G.flying){
    drawSky();drawPlayer();
  } else {
    var evIds=EVTS.map(function(e){return e.id;});
    if(G.world==='hub')drawHub();
    else if(G.world==='dungeon')drawDungeon();
    else if(G.world==='special')drawSpecial();
    else if(G.world==='maze')drawMaze();
    else if(evIds.indexOf(G.world)>=0)drawEvWorld(G.world);
    else drawHub();
    if(G.world!=='maze'){drawMons();drawOnlinePlayers();drawPlayer();}
  }
  hud();
  requestAnimationFrame(loop);
}
